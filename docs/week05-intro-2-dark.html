課堂練習 1：變數 (variables)、溢位 (overflow)

教學影片傳送門：https://youtu.be/zwHdffsiKuY?t=117

#include <iostream>
#include <climits>

using namespace std;

int main()
{
    int p, n;

    cout << "sizeof(int) = " << sizeof(int) << endl;
    cout << "Max Integer = " << INT_MAX << endl;
    cout << "Min Integer = " << INT_MIN << endl;

    p = INT_MAX;
    n = INT_MIN;
    cout << "p = " << p << endl;
    cout << "n = " << n << endl;
    cout << "(p + 1) = " << (p + 1) << endl;
    cout << "(n - 1) = " << (n - 1) << endl;

    return 0;
}

課堂練習 3：數值運算 (+、-、*、/、 %)

教學影片傳送門：https://youtu.be/zwHdffsiKuY?t=848

#include <iostream>

using namespace std;

int main()
{
    int a, b;

    a = 123;
    b = 100;

    cout << "a = " << a << endl;
    cout << "b = " << b << endl;
    cout << "(a + b)        = " << (a + b) << endl;
    cout << "(a - b)        = " << (a - b) << endl;
    cout << "(a * b)        = " << (a * b) << endl;
    cout << "(a / b)        = " << (a / b) << endl;
    cout << "(a / 100.0)    = " << (a / 100.0) << endl;
    cout << "(a / (float)b) = " << (a / (float)b) << endl;
    cout << "(a % 100)      = " << (a % 100) << endl;

    return 0;
}

課堂練習 4：關係運算 (>、>=、<、<=、==、!=) 、 連接 ( &&、||)

教學影片傳送門：https://youtu.be/zwHdffsiKuY?t=1186

#include <iostream>

using namespace std;

int main()
{
    int a = 120;
    int b = 100;

    cout << "a        = " << a << endl;
    cout << "b        = " << b << endl;
    cout << "(a > b)  = " << (a > b) << endl;
    cout << "(a < b)  = " << (a < b) << endl;
    cout << "(a >= a) = " << (a >= a) << endl;
    cout << "(a <= a) = " << (a <= a) << endl;
    cout << "(a == b) = " << (a == b) << endl;
    cout << "(b == b) = " << (b == b) << endl;
    cout << "(a != b) = " << (a != b) << endl;
    cout << "(b != a) = " << (b != a) << endl;
    cout << "(a % 4 == 0) = " << (a % 4 == 0) << endl;
    cout << "(a % 3 != 0) = " << (a % 3 != 0) << endl;
    cout << "(b % 4 == 0) = " << (b % 4 == 0) << endl;
    cout << "(b % 3 == 0) = " << (b % 3 == 0) << endl;
    cout << "(a % 4 == 0) && (a % 3 != 0) = "
         << ((a % 4 == 0) && (a % 3 != 0)) << endl;
    cout << "(b % 4 == 0) || (b % 3 == 0) = "
         << ((b % 4 == 0) || (b % 3 == 0)) << endl;

    return 0;
}

課堂練習 6：for 迴圈應用 - 重複 (repetition)

課堂練習 7：for 迴圈應用 - 迭代 (iteration)

教學影片傳送門：https://youtu.be/zwHdffsiKuY?t=2039

#include <iostream>

using namespace std;

int main()
{
    int n;
    cout << "Enter an integer (n) : ";
    cin >> n;

    for (int i = 1; i <= n; i++)
    {
        cout << i << " x " << i << " = " << i * i << endl;
    }
    cout << endl;

    return 0;
}

課堂練習 8：for 迴圈應用 - 不同方向的迭代

教學影片傳送門：https://youtu.be/zwHdffsiKuY?t=2211

#include <iostream>
#include <iomanip>

using namespace std;

int main()
{
    int n;
    cout << "Enter an integer (n) : ";
    cin >> n;

    for (int i = 1; i <= n; i++)
    {
        cout << setw(4 + i) << i << " *********" << endl;
    }
    for (int i = n - 1; i > 0; i--)
    {
        cout << setw(4 + i) << i << " *********" << endl;
    }
    cout << endl;

    return 0;
}
